alter table slots enable row level security; create policy "Slots públicos" on slots for select using (true); create policy "Propietario crea slots" on slots for insert with check (auth.uid() is not null); -- POSTS create table if not exists posts ( id bigserial primary key, user_id uuid references profiles(id) on delete cascade, username text, avatar_url text, restaurant_id bigint references restaurants(id), restaurant_name text, img text not null, type text default 'photo', caption text, likes_count int default 0, created_at timestamptz default now() ); alter table likes enable row level security; create policy "Likes públicos" on likes for select using (true); create policy "Usuarios dan like" on likes for insert with check (auth.uid() = user_id); create policy "Usuarios quitan like" on likes for delete using (auth.uid() = user_id); -- FOLLOWS restaurantes create table if not exists follows ( user_id uuid references profiles(id) on delete cascade, restaurant_id bigint references restaurants(id) on delete cascade, created_at timestamptz default now(), primary key(user_id, restaurant_id) ); alter table follows enable row level security; create policy "Follows públicos" on follows for select using (true); create policy "Usuarios siguen" on follows for insert with check (auth.uid() = user_id); create policy "Usuarios dejan de seguir" on follows for delete using (auth.uid() = user_id); -- MESSAGES create table if not exists messages ( id bigserial primary key, from_id uuid references profiles(id) on delete cascade, to_id uuid references profiles(id) on delete cascade, from_username text, text text not null, read boolean default false, created_at timestamptz default now() ); alter table messages enable row level security; create policy "Ver mis mensajes" on messages for select using (auth.uid() = from_id or auth.uid() = to_id); create policy "Enviar mensajes" on messages for insert with check (auth.uid() = from_id);
